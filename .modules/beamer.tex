\makeatletter

% Report an error when leaded in a non-Beamer document.
\@ifclassloaded{beamer}{}{
    \PackageError{beamer.tex}{Loaded in a non-Beamer document}
}

% Theme and font
\usefonttheme{serif}
\usecolortheme{metropolis}

% Remove document icons in the bibliography
\setbeamertemplate{bibliography item}[text]

% No color for links
\hypersetup{colorlinks=false}

% Cite with "&" instead of "and"
\renewcommand*{\finalnamedelim}{
    \ifnumgreater{\value{liststop}}{2}{\finalandcomma}{}\addspace\&\space
}
% Note that this also affects the bibliography.

% Beamer won't display footnotes unless `[frame]` is specified.
\let\oldfootnote\footnote
\renewcommand{\footnote}[1]{\oldfootnote[frame]{#1}}
\@ifpackageloaded{biblatex-chicago}{
    \renewcommand{\footcite}[1]{\oldfootnote[frame]{\cite{#1}}}
}{}

\makeatother
