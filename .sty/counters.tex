\MakePerPage{equation}

\MakePerPage{theorem}
\renewcommand{\thetheorem}{\theperpage.\arabic{theorem}}

\MakeSortedPerPage{figure}
\renewcommand{\thefigure}{\theperpage.\arabic{figure}}

\MakeSortedPerPage{table}
\renewcommand{\thetable}{\theperpage.\arabic{table}}
