% In theory `hyperref` should be loaded last (but before `cleveref`), but the
% current setup works, and I do not want to change it.
\usepackage[]{hyperref}
\hypersetup{
    allcolors = violet,
    breaklinks = true,
    colorlinks = true,
    hypertexnames = false, % or `perpage` will lead to duplicate PDF anchors
}

% Load after `hyperref` and `amsmath` but before definitions of theorem-like
% environments. Not compatible with `beamer`.
\makeatletter
\@ifclassloaded{beamer}{}{
    \usepackage[capitalize, noabbrev]{cleveref}
}
\makeatother

% Set links in roman
\urlstyle{rm}
