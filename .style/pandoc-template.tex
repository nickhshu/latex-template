% A minimal Pandoc template for R Markdown output.

% The default template can be found at
% <https://github.com/jgm/pandoc/blob/master/data/templates/default.latex>.
% Note that the the default template also imports other templates (e.g.,
% `common.latex` and `fonts.latex`). Consult these files when necessary.

\documentclass{article}

\usepackage{import}

\import{.style/}{normal.tex}
% TODO: usually needs to be replaced with the absolute path

\makeatletter

% No page numbers (useful if the output will be attached to another document)
% \pagenumbering{gobble}

% Smaller font for code
\setmonofont[Scale=0.7]{Fira Mono}

% Add space between paragraphs and remove indentation.
\usepackage[parfill]{parskip}

% The rest are lifted from the default template.
$if(highlighting-macros)$ % chktex 46
$highlighting-macros$ % chktex 46
$endif$ % chktex 46

\providecommand{\tightlist}{\setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

\usepackage{graphicx}
\def\maxwidth{\ifdim\Gin@nat@width>\linewidth\linewidth\else\Gin@nat@width\fi}
\def\maxheight{\ifdim\Gin@nat@height>\textheight\textheight\else\Gin@nat@height\fi}
\setkeys{Gin}{width=\maxwidth,height=\maxheight,keepaspectratio}
\def\fps@figure{htbp}

\newsavebox\pandoc@box
\newcommand*\pandocbounded[1]{% scales image to fit in text height/width
  \sbox\pandoc@box{#1}%
  \Gscale@div\@tempa{\textheight}{\dimexpr\ht\pandoc@box+\dp\pandoc@box\relax}%
  \Gscale@div\@tempb{\linewidth}{\wd\pandoc@box}%
  \ifdim\@tempb\p@<\@tempa\p@\let\@tempa\@tempb\fi% select the smaller of both
  \ifdim\@tempa\p@<\p@\scalebox{\@tempa}{\usebox\pandoc@box}%
  \else\usebox{\pandoc@box}%
  \fi%
}

\makeatother

\title{$title$} % chktex 46
\date{}

\begin{document}

\maketitle

$body$ % chktex 46

\end{document}
