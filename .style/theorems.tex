% Style `it` is the default style that sets the body in italics.
\newtheoremstyle{it}% name
{}% space above
{}% space below
{\itshape}% body font
{}% indent
{\bfseries}% head font
{}% head punctuation
{0.5em}% after head space
{\thmname{#1}\thmnumber{ #2}\thmnote{ (#3)}.}% head spec

% Style `rmnl` sets the body in roman and breaks line after the header.
\newtheoremstyle{rmnl}% name
{}% space above
{}% space below
{\mdseries}% body font
{}% indent
{\bfseries}% head font
{}% head punctuation
{\newline}% after head space
{\thmname{#1}\thmnumber{ #2}\thmnote{: #3}}% head spec

% -----------------------------------------------------------------------------
\theoremstyle{it}

\newtheorem{theorem}{Theorem}
\newtheorem*{theorem*}{Theorem}
\BeforeBeginEnvironment{theorem}{\goodbreak}
\BeforeBeginEnvironment{theorem*}{\goodbreak}
% The use of `\goodbreak` works around incorrect page-based numbering. See
% <https://texfaq.org/FAQ-wrongpn> and <https://tex.stackexchange.com/a/88186>.

\newcommand{\newthm}[2]{%
    \newtheorem{#1}[theorem]{#2}
    \newtheorem*{#1*}{#2}
    \BeforeBeginEnvironment{#1}{\goodbreak}
    \BeforeBeginEnvironment{#1*}{\goodbreak}
}

\newthm{lemma}{Lemma}
\newthm{corollary}{Corollary}
\newthm{claim}{Claim}
\newthm{proposition}{Proposition}

\newthm{definition}{Definition}
\newthm{example}{Example}
\newthm{procedure}{Procedure}
\newthm{remark}{Remark}
