% Theoretically `hyperref` should be loaded last (except for `cleveref`).
% However the current setup works, and I do not intend to change it.
\usepackage[]{hyperref}
\hypersetup{
    allcolors = violet,
    breaklinks = true,
    colorlinks = true,
    hypertexnames = false, % Otherwise `perpage` will lead to duplicate anchors
}

% Load after `hyperref` and `amsmath` but before definitions of proclamation
% environments. Do not load with `beamer`. They are not compatible.
\makeatletter
\@ifclassloaded{beamer}{}{
    \usepackage[capitalize, noabbrev]{cleveref}
}
\makeatother
